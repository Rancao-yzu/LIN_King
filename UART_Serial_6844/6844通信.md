# 6844通信

### 1、6844串口协议header

totalPacketLen：包含串口header信息，frame header和frame pointcloud三类数据的长度，是按32bytes对齐的，发送32字节倍数，padding数据为0.

```c
typedef struct MmwDemo_output_message_headerID_t {
    uint16_t magicWord[4];   // 固定同步字: 0x0102, 0x0304, 0x0506, 0x0708
    uint32_t version;        // SDK版本号
    uint32_t totalPacketLen; // 整个包的总字节数（32字节对齐）
    uint32_t platform;       // 平台类型，这里写死 0xA6432
    uint32_t frameNumber;    // 帧序号，从0开始递增
    uint32_t timeCpuCycles;  // 时间戳（CPU周期数）
    uint16_t numDetectedObj[2]; // [0]=当前帧点云数量，[1]=预留为0
    uint32_t numTLVs;        // TLV结构数量（这里是1）
    uint32_t subFrameNumber; // 子帧号，非高级帧配置下为0
} MmwDemo_output_message_headerID;

```
解析时首先验证同步字，四个 magicWord 必须匹配 0x0102, 0x0304, 0x0506, 0x0708，否则说明数据错位。


### 2、6844 frame header

Length: 包含串口frame header和frame pointcloud两类数据的长度

MmwDemo_output_message_point_unit：点云数据单位结构体

```c
typedef struct MmwDemo_output_message_tl_t {
    uint32_t type;   // 固定为 MMWDEMO_OUTPUT_MSG_POINT_CLOUD
    uint32_t length; // Frame Header + 所有点云数据的字节数
} MmwDemo_output_message_tl;

typedef struct MmwDemo_output_message_point_uint_t {
    float elevationUnit; // 俯仰角单位：弧度/量化值
    float azimuthUnit;   // 方位角单位：弧度/量化值
    float dopplerUnit;   // 多普勒速度单位：m/s 每量化值
    float rangeUnit;     // 距离单位：m 每量化值
    float snrUint;       // SNR单位：线性值/量化值
} MmwDemo_output_message_point_unit;

```

### 3、6844点云协议

```c
typedef struct MmwDemo_output_message_UARTpoint_t {
    int8_t   elevation;  // 俯仰角量化值（有符号）
    int8_t   azimuth;    // 方位角量化值（有符号）
    int16_t  doppler;    // 多普勒速度量化值
    uint16_t range;      // 距离量化值
    uint16_t snr;        // SNR量化值
} MmwDemo_output_message_UARTpoint;


typedef struct MmwDemo_output_message_UARTpointCloud_t
{
    MmwDemo_output_message_tl           messageTL;     // TLV头部
    MmwDemo_output_message_point_unit   pointUnit;     // 单位定义
    MmwDemo_output_message_UARTpoint    point[MMWDEMO_OUTPUT_POINT_CLOUD_LIST_MAX_SIZE]; // 点云数组

} MmwDemo_output_message_UARTpointCloud;

```


### 4、点云单位的换算逻辑

```c
/* Header: */
headerID.platform =  0xA6432;
headerID.magicWord[0] = 0x0102;
headerID.magicWord[1] = 0x0304;
headerID.magicWord[2] = 0x0506;
headerID.magicWord[3] = 0x0708;
headerID.numDetectedObj[0] = gMmwMssMCB.numDetectedPoints;
headerID.numDetectedObj[1] = 0;
headerID.version =    MMWAVE_SDK_VERSION_BUILD |   //DEBUG_VERSION
                    (MMWAVE_SDK_VERSION_BUGFIX << 8) |
                    (MMWAVE_SDK_VERSION_MINOR << 16) |
                    (MMWAVE_SDK_VERSION_MAJOR << 24);

headerID.numTLVs = tlvIdx;
/* Round up packet length to multiple of MMWDEMO_OUTPUT_MSG_SEGMENT_LEN */
headerID.totalPacketLen = MMWDEMO_OUTPUT_MSG_SEGMENT_LEN *
        ((packetLen + (MMWDEMO_OUTPUT_MSG_SEGMENT_LEN-1))/MMWDEMO_OUTPUT_MSG_SEGMENT_LEN);

headerID.timeCpuCycles =  0; //TODO: Populate with actual time
headerID.frameNumber = gMmwMssMCB.stats.frameStartIntCounter;
headerID.subFrameNumber = -1;


gMmwMssMCB.pointCloudToUart.messageTL.type             = MMWDEMO_OUTPUT_MSG_POINT_CLOUD;
gMmwMssMCB.pointCloudToUart.pointUnit.azimuthUnit   = (PI/2.f)/127.f;
gMmwMssMCB.pointCloudToUart.pointUnit.elevationUnit = (PI/2.f)/127.f;
gMmwMssMCB.pointCloudToUart.pointUnit.rangeUnit     = 0.00025f;
gMmwMssMCB.pointCloudToUart.pointUnit.dopplerUnit   = 0.00028f;
gMmwMssMCB.pointCloudToUart.pointUnit.snrUint       = 1./256.;//snr from DSP is in dB in Q8 format

gMmwMssMCB.pointUnitInv.azimuthUnit    = 1./gMmwMssMCB.pointCloudToUart.pointUnit.azimuthUnit;
gMmwMssMCB.pointUnitInv.elevationUnit  = 1./gMmwMssMCB.pointCloudToUart.pointUnit.elevationUnit;
gMmwMssMCB.pointUnitInv.rangeUnit      = 1./gMmwMssMCB.pointCloudToUart.pointUnit.rangeUnit;
gMmwMssMCB.pointUnitInv.dopplerUnit    = 1./gMmwMssMCB.pointCloudToUart.pointUnit.dopplerUnit;
gMmwMssMCB.pointUnitInv.snrUnit        = 1./gMmwMssMCB.pointCloudToUart.pointUnit.snrUnit;

```
